```python
from typing import Any, Optional
from ..core.config import ConfigManager
from ..db.models import Setting
from ..db.session import get_session

class SettingsService:
    """Service layer for settings management, encapsulating Core ConfigManager"""

    @staticmethod
    def get_setting(key: str) -> Optional[str]:
        """Get a setting value by key"""
        with get_session() as session:
            setting = session.get(Setting, key)
            return setting.value if setting else None

    @staticmethod
    def set_setting(key: str, value: str) -> None:
        """Set a setting value"""
        with get_session() as session:
            setting = session.get(Setting, key)
            if setting:
                setting.value = value
            else:
                setting = Setting(key=key, value=value)
                session.add(setting)
            session.commit()

    @staticmethod
    def get_all_settings() -> dict[str, str]:
        """Get all settings as dict"""
        with get_session() as session:
            settings = session.query(Setting).all()
            return {s.key: s.value for s in settings}

    # Add methods that wrap ConfigManager if needed
    @staticmethod
    def get_config_manager() -> ConfigManager:
        """Get ConfigManager instance for advanced usage"""
        return ConfigManager()
```

```python
from typing import Optional
from ..core import metadata as metadata_module

class MetadataService:
    """Service layer for metadata extraction, encapsulating Core metadata module"""

    @staticmethod
    def get_media_info(media_path: str) -> Optional[dict]:
        """Extract media info from NFO/TXT files"""
        return metadata_module.get_media_info(media_path)

    @staticmethod
    def extract_nfo_data(nfo_path: str) -> Optional[dict]:
        """Extract metadata from NFO file"""
        return metadata_module.extract_nfo_data(nfo_path)

    @staticmethod
    def extract_poster(poster_dir: str) -> Optional[str]:
        """Find poster image path"""
        return metadata_module.extract_poster(poster_dir)

    @staticmethod
    def extract_fallback_metadata(txt_dir: str) -> Optional[dict]:
        """Extract fallback metadata from TXT"""
        return metadata_module.extract_fallback_metadata(txt_dir)
```
